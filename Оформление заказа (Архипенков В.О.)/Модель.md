@startuml
class User {
  - id: Long
  - name: String
  - email: String
  - phone: String
  - balance: Decimal
  - type: UserType
  + createOrder()
  + makePayment()
  + sendMessage()
}

class Order {
  - id: Long
  - status: OrderStatus
  - totalAmount: Decimal
  - commission: Decimal
  - creationDate: DateTime
  + calculateTotal()
  + updateStatus()
  + cancelOrder()
}

class DeliverySystem {
  - availableServices: List<DeliveryService>
  + calculateCost()
  + scheduleDelivery()
  + trackDelivery()
  + updateStatus()
}

class PaymentSystem {
  - supportedMethods: List<PaymentMethod>
  + processPayment()
  + confirmTransaction()
  + refundPayment()
  + generateReceipt()
}

class NotificationSystem {
  - notificationTypes: List<NotificationType>
  + sendSMS()
  + sendEmail()
  + sendPush()
  + updateStatus()
}


' Relationships
User "1" -- "*" Order : places
Order "1" -- "1" DeliverySystem : uses
Order "1" -- "1" PaymentSystem : processes through
Order "1" -- "1" NotificationSystem : notifies through
User "1" -- "*" NotificationSystem : receives

@enduml
